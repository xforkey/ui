### Collapsible Code Example

The `CodeExample` component can be made collapsible by setting the `collapsible` prop to `true`.


<CodeExample collapsible={true} example={tsx`
<CodeExample collapsible={true} example={tsx\`
import React from 'react';

function ComplexComponent() {
  const [count, setCount] = React.useState(0);
  
  const increment = () => {
    setCount(prevCount => prevCount + 1);
  };
  
  const decrement = () => {
    setCount(prevCount => prevCount - 1);
  };
  
  return (
    <div className="p-4 border rounded-md">
      <h2 className="text-lg font-bold">Counter: {count}</h2>
      <div className="flex space-x-2 mt-2">
        <button 
          className="px-3 py-1 bg-blue-500 text-white rounded-md hover:bg-blue-600"
          onClick={increment}
        >
          Increment
        </button>
        <button 
          className="px-3 py-1 bg-red-500 text-white rounded-md hover:bg-red-600"
          onClick={decrement}
        >
          Decrement
        </button>
      </div>
    </div>
  );
}

export default ComplexComponent;
\`} />
`} />
